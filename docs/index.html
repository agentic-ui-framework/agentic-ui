Complete Integration Example

// Your existing Next.js app
import { Agentic UIProvider, Agentic UIChat } from '@agentic/frabrik'

export default function App() {
  return (
    <Agentic UIProvider
      apiKey={process.env.FRABRIK_API_KEY}
      tenant={{
        id: 'your-company',
        branding: yourBrand
      }}
      connectors={{
        database: { type: 'postgresql', url: dbUrl },
        api: { type: 'graphql', endpoint: apiUrl }
      }}
    >
      {/* Your existing app */}
      <YourSaaSInterface />
      
      {/* Add Agentic UI magic */}
      <Agentic UIChat position="bottom-right" />
    </Agentic UIProvider>
  )
}

Integrate in 5 Minutes

Add Agentic UI to your SaaS and watch retention soar

01

Install Agentic UI SDK

npm install @frabrik/sdk
02

Connect to Your SaaS Data

const frabrik = new Agentic UI({
  // Connect your existing database
  connectors: {
    database: {
      type: 'postgresql',
      connection: process.env.DATABASE_URL,
      schema: 'public'
    },
    // Or connect your APIs
    api: {
      type: 'rest',
      baseUrl: 'https://api.yourapp.com'
    }
  }
})
PostgreSQL
MySQL
MongoDB
REST/GraphQL
03

Enable for Your Customers

frabrik.configure({
  // Each of YOUR customers gets their own space
  tenant: {
    id: 'customer-uuid',
    name: 'Acme Corp',
    // White-label branding per tenant
    branding: {
      logo: 'customer.com/logo.png',
      primaryColor: '#FF6B00'
    },
    // Data access permissions
    permissions: {
      tables: ['customers', 'orders'],
      operations: [
        'read', 'write', 'create'
      ]
    }
  }
})
04

Setup Authentication

frabrik.auth({
  // Use your existing auth system
  provider: 'oauth2',
  issuer: 'https://auth.yourapp.com',
  // Map user roles to Agentic UI permissions
  roleMapping: {
    admin: ['create', 'edit', 'delete'],
    user: ['create', 'edit'],
    viewer: ['view']
  }
})
JWT Auth OAuth 2.0 SAML RBAC

Add Agentic UI Interface

// Add to your React app
<Agentic UIChat
  user={currentUser}
  permissions={userPermissions}
  theme={yourBrandTheme}
/>
