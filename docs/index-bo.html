Quick Start Code 
// 1. Install via npm or CDN
npm install @agentic-ui/agent

// 2. Initialize in your app
import { AgenticUI } from '@agentic-ui/agent';

const agent = new AgenticUI({
  apiKey: 'sk_live_...',
  mode: 'production',

  // Connect to your backend
  integrations: {
    salesforce: {
      instanceUrl: 'https://your.salesforce.com',
      accessToken: process.env.SF_TOKEN
    },
    sap: {
      endpoint: 'https://api.sap.com/your-instance',
      credentials: process.env.SAP_CREDS
    }
  },

  // Define automation rules
  rules: {
    'employee-requests': {
      trigger: 'form.submit',
      conditions: ['type == "IT_request"'],
      action: 'auto-approve-if-standard'
    }
  }
});

// 3. Agent is now active
agent.start();
